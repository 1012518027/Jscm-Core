// 手动定义 NT_SUCCESS
#ifndef NT_SUCCESS
#define NT_SUCCESS(Status) (((NTSTATUS)(Status)) >= 0)
#endif

// 提升 SeDebugPrivilege
bool EnableDebugPrivilege() {
    typedef NTSTATUS(NTAPI* pRtlAdjustPrivilege)(
        ULONG Privilege,
        BOOLEAN Enable,
        BOOLEAN CurrentThread,
        PBOOLEAN OldValue
        );
    HMODULE hNtDll = GetModuleHandleA("ntdll.dll");
    if (!hNtDll) return false;

    pRtlAdjustPrivilege RtlAdjustPrivilege =
        (pRtlAdjustPrivilege)GetProcAddress(hNtDll, "RtlAdjustPrivilege");
    if (RtlAdjustPrivilege) {
        BOOLEAN oldValue;
        NTSTATUS status = RtlAdjustPrivilege(20, TRUE, FALSE, &oldValue); // SE_DEBUG_PRIVILEGE = 20
        return NT_SUCCESS(status);
    }
    return false;
}

// 通过进程名获取 PID（已修复 Unicode 问题）
DWORD GetProcessIdByName(const std::string& processName) {
    std::wstring wProcessName(processName.begin(), processName.end());

    HANDLE hSnapshot = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    if (hSnapshot == INVALID_HANDLE_VALUE) return 0;

    PROCESSENTRY32 pe32{};
    pe32.dwSize = sizeof(pe32);

    if (Process32First(hSnapshot, &pe32)) {
        do {
            if (_wcsicmp(pe32.szExeFile, wProcessName.c_str()) == 0) {
                CloseHandle(hSnapshot);
                return pe32.th32ProcessID;
            }
        } while (Process32Next(hSnapshot, &pe32));
    }

    CloseHandle(hSnapshot);
    return 0;
}

// 打印 32 位寄存器
void PrintRegisters(const CONTEXT& context) {
    std::cout << "\n===== 32位寄存器状态 =====\n";
    std::cout << "EAX: 0x" << std::hex << std::uppercase << std::setw(8) << std::setfill('0') << context.Eax << " ";
    std::cout << "EBX: 0x" << std::setw(8) << context.Ebx << "\n";
    std::cout << "ECX: 0x" << std::setw(8) << context.Ecx << " ";
    std::cout << "EDX: 0x" << std::setw(8) << context.Edx << "\n";
    std::cout << "ESI: 0x" << std::setw(8) << context.Esi << " ";
    std::cout << "EDI: 0x" << std::setw(8) << context.Edi << "\n";
    std::cout << "ESP: 0x" << std::setw(8) << context.Esp << " ";
    std::cout << "EBP: 0x" << std::setw(8) << context.Ebp << "\n";
    std::cout << "EIP: 0x" << std::setw(8) << context.Eip << "\n\n";

    std::cout << "===== 段寄存器 =====\n";
    std::cout << "CS: 0x" << std::hex << std::setw(4) << context.SegCs << " ";
    std::cout << "DS: 0x" << std::setw(4) << context.SegDs << "\n";
    std::cout << "ES: 0x" << std::setw(4) << context.SegEs << " ";
    std::cout << "FS: 0x" << std::setw(4) << context.SegFs << "\n";
    std::cout << "GS: 0x" << std::setw(4) << context.SegGs << " ";
    std::cout << "SS: 0x" << std::setw(4) << context.SegSs << "\n\n";

    std::cout << "===== 标志寄存器 =====\n";
    std::cout << "EFlags: 0x" << std::hex << std::setw(8) << context.EFlags << "\n\n";
    std::cout << std::dec << std::nouppercase;
}

// 设置硬件断点（支持读/写/读写触发）
bool SetHardwareBreakpoint(HANDLE hThread, DWORD address, int triggerType, int size) {
    CONTEXT context = { 0 };
    context.ContextFlags = CONTEXT_DEBUG_REGISTERS;

    if (!GetThreadContext(hThread, &context)) {
        std::cerr << "GetThreadContext 失败: " << GetLastError() << "\n";
        return false;
    }

    // 查找空闲调试寄存器 DR0-DR3
    int regIndex = -1;
    for (int i = 0; i < 4; ++i) {
        if (!(context.Dr7 & (1ULL << (i * 2)))) {
            regIndex = i;
            break;
        }
    }
    if (regIndex == -1) {
        std::cerr << "无可用调试寄存器（最多4个）\n";
        return false;
    }

    // 设置断点地址
    switch (regIndex) {
    case 0: context.Dr0 = address; break;
    case 1: context.Dr1 = address; break;
    case 2: context.Dr2 = address; break;
    case 3: context.Dr3 = address; break;
    }

    // 启用该断点（局部启用）
    context.Dr7 |= (1ULL << (regIndex * 2));

    // 设置触发条件
    // triggerType: 1 = 写入, 3 = 读或写（用于“读取”和“读写”）
    int rwBits = (triggerType == 1) ? 1 : 3;  // 1=写, 3=读写
    context.Dr7 &= ~(0x3ULL << (16 + regIndex * 4));
    context.Dr7 |= ((ULONGLONG)rwBits << (16 + regIndex * 4));

    // 设置长度
    int lenCode = 0;
    if (size == 1) lenCode = 0;
    else if (size == 2) lenCode = 1;
    else if (size == 4) lenCode = 3;
    else {
        std::cerr << "不支持的长度，仅支持 1、2、4 字节\n";
        return false;
    }
    context.Dr7 &= ~(0x3ULL << (18 + regIndex * 4));
    context.Dr7 |= ((ULONGLONG)lenCode << (18 + regIndex * 4));

    if (!SetThreadContext(hThread, &context)) {
        std::cerr << "SetThreadContext 失败: " << GetLastError() << "\n";
        return false;
    }

    const char* typeName = (triggerType == 1) ? "写入触发" : "读或写触发";
    std::cout << "[+] 在 DR" << regIndex << " 设置硬件断点成功：\n";
    std::cout << "    地址: 0x" << std::hex << address << "\n";
    std::cout << "    类型: " << typeName << "\n";
    std::cout << "    长度: " << std::dec << size << " 字节\n\n";
    return true;
}

// 处理调试事件
void HandleDebugEvent(const DEBUG_EVENT& debugEvent) {
    if (debugEvent.dwDebugEventCode == EXCEPTION_DEBUG_EVENT) {
        const EXCEPTION_RECORD& er = debugEvent.u.Exception.ExceptionRecord;
        if (er.ExceptionCode == EXCEPTION_SINGLE_STEP) {
            std::cout << "\n>>> 硬件断点触发！<<<\n";
            std::cout << "触发线程: " << debugEvent.dwThreadId << "\n";
            std::cout << "触发地址: 0x" << std::hex << (DWORD)(uintptr_t)er.ExceptionAddress << "\n\n";

            HANDLE hThread = OpenThread(THREAD_GET_CONTEXT, FALSE, debugEvent.dwThreadId);
            if (hThread) {
                CONTEXT ctx = { 0 };
                ctx.ContextFlags = CONTEXT_FULL | CONTEXT_i386;
                if (GetThreadContext(hThread, &ctx)) {
                    PrintRegisters(ctx);
                }
                else {
                    std::cerr << "读取线程上下文失败: " << GetLastError() << "\n";
                }
                CloseHandle(hThread);
            }
            else {
                std::cerr << "OpenThread 失败: " << GetLastError() << "\n";
            }
        }
    }
}

int main() {
    if (!EnableDebugPrivilege()) {
        std::cerr << "警告：提升调试权限失败（请以管理员身份运行）\n";
    }

    std::cout << "=== x86 硬件断点调试器（支持读/写/读写触发）===\n";
    std::cout << "输入目标进程名（如 notepad.exe）: ";
    std::string procName;
    std::getline(std::cin, procName);

    DWORD pid = GetProcessIdByName(procName);
    if (pid == 0) {
        std::cerr << "错误：未找到进程 \"" << procName << "\"\n";
        return 1;
    }
    std::cout << "[+] 找到进程 PID: " << pid << "\n";

    if (!DebugActiveProcess(pid)) {
        std::cerr << "DebugActiveProcess 失败: " << GetLastError() << "\n";
        return 1;
    }
    std::cout << "[+] 已成功附加到进程\n";

    // 获取任意一个线程用于设置断点
    HANDLE hSnap = CreateToolhelp32Snapshot(TH32CS_SNAPTHREAD, 0);
    DWORD targetTid = 0;
    if (hSnap != INVALID_HANDLE_VALUE) {
        THREADENTRY32 te{ sizeof(te) };
        if (Thread32First(hSnap, &te)) {
            do {
                if (te.th32OwnerProcessID == pid) {
                    targetTid = te.th32ThreadID;
                    break;
                }
            } while (Thread32Next(hSnap, &te));
        }
        CloseHandle(hSnap);
    }

    if (targetTid == 0) {
        std::cerr << "无法获取目标进程的线程\n";
        DebugActiveProcessStop(pid);
        return 1;
    }

    HANDLE hThread = OpenThread(THREAD_GET_CONTEXT | THREAD_SET_CONTEXT | THREAD_SUSPEND_RESUME, FALSE, targetTid);
    if (!hThread) {
        std::cerr << "OpenThread 失败: " << GetLastError() << "\n";
        DebugActiveProcessStop(pid);
        return 1;
    }

    // 用户选择触发类型
    std::cout << "\n请选择触发类型：\n";
    std::cout << "1. 读取触发（实际为读或写）\n";
    std::cout << "2. 写入触发（仅写）\n";
    std::cout << "3. 读写触发（读或写，和选项1相同）\n> ";
    int choice;
    std::cin >> choice;

    int triggerType = 3; // 默认读写
    if (choice == 2) {
        triggerType = 1; // 纯写
    }
    // choice 1 和 3 都映射为 3（读或写）

    std::cin.ignore(); // 清除换行符

    std::cout << "输入监控地址（十六进制，如 00401000）: 0x";
    DWORD addr;
    std::cin >> std::hex >> addr;

    std::cout << "输入监控长度（1/2/4 字节）: ";
    int len;
    std::cin >> len;

    if (!SetHardwareBreakpoint(hThread, addr, triggerType, len)) {
        CloseHandle(hThread);
        DebugActiveProcessStop(pid);
        return 1;
    }

    CloseHandle(hThread);

    std::cout << "[+] 断点设置完成！程序将持续监控，直到触发或你手动终止（Ctrl+C）\n\n";

    // 主调试循环
    DEBUG_EVENT de;
    while (WaitForDebugEvent(&de, INFINITE)) {
        HandleDebugEvent(de);
        ContinueDebugEvent(de.dwProcessId, de.dwThreadId, DBG_CONTINUE);
    }

    DebugActiveProcessStop(pid);
    std::cout << "\n[+] 调试已结束。\n";
    return 0;
}
