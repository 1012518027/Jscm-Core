// 手动定义 NT_SUCCESS（避免依赖 winternl.h）
#ifndef NT_SUCCESS
#define NT_SUCCESS(Status) (((NTSTATUS)(Status)) >= 0)
#endif

// 提升调试权限
bool EnableDebugPrivilege() {
    typedef NTSTATUS(NTAPI* pRtlAdjustPrivilege)(
        ULONG Privilege,
        BOOLEAN Enable,
        BOOLEAN CurrentThread,
        PBOOLEAN OldValue
        );

    HMODULE hNtDll = GetModuleHandleA("ntdll.dll"); // 使用 ANSI 版本
    if (!hNtDll) return false;

    pRtlAdjustPrivilege RtlAdjustPrivilege =
        (pRtlAdjustPrivilege)GetProcAddress(hNtDll, "RtlAdjustPrivilege");

    if (RtlAdjustPrivilege) {
        BOOLEAN tmp1;
        NTSTATUS status = RtlAdjustPrivilege(20, TRUE, FALSE, &tmp1); // SE_DEBUG_PRIVILEGE
        return NT_SUCCESS(status);
    }
    return false;
}

// 通过进程名获取 PID（兼容所有字符集）
DWORD GetProcessIdByName(const std::string& processName) {
    // 强制使用 ANSI 版本（避免宽字符兼容问题）
    PROCESSENTRY32 pe32 = { 0 };
    pe32.dwSize = sizeof(PROCESSENTRY32);

    HANDLE hSnapshot = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    if (hSnapshot == INVALID_HANDLE_VALUE) return 0;

    if (Process32First(hSnapshot, &pe32)) {
        do {
            // 直接使用 ANSI 比较（项目字符集设为多字节）
            if (_stricmp(pe32.szExeFile, processName.c_str()) == 0) {
                CloseHandle(hSnapshot);
                return pe32.th32ProcessID;
            }
        } while (Process32Next(hSnapshot, &pe32));
    }
    CloseHandle(hSnapshot);
    return 0;
}

// 打印寄存器（x64）
void PrintRegisters(const CONTEXT& context) {
    std::cout << "\n===== 寄存器状态 =====\n";
    std::cout << "RAX: 0x" << std::hex << std::setw(16) << std::setfill('0') << context.Rax << "  ";
    std::cout << "RBX: 0x" << std::hex << std::setw(16) << std::setfill('0') << context.Rbx << "\n";
    std::cout << "RCX: 0x" << std::hex << std::setw(16) << std::setfill('0') << context.Rcx << "  ";
    std::cout << "RDX: 0x" << std::hex << std::setw(16) << std::setfill('0') << context.Rdx << "\n";
    std::cout << "RSI: 0x" << std::hex << std::setw(16) << std::setfill('0') << context.Rsi << "  ";
    std::cout << "RDI: 0x" << std::hex << std::setw(16) << std::setfill('0') << context.Rdi << "\n";
    std::cout << "RSP: 0x" << std::hex << std::setw(16) << std::setfill('0') << context.Rsp << "  ";
    std::cout << "RBP: 0x" << std::hex << std::setw(16) << std::setfill('0') << context.Rbp << "\n";
    std::cout << "RIP: 0x" << std::hex << std::setw(16) << std::setfill('0') << context.Rip << "\n\n";

    std::cout << "===== 段寄存器 =====\n";
    std::cout << "CS:  0x" << std::hex << std::setw(4) << std::setfill('0') << context.SegCs << "  ";
    std::cout << "DS:  0x" << std::hex << std::setw(4) << std::setfill('0') << context.SegDs << "\n";
    std::cout << "ES:  0x" << std::hex << std::setw(4) << std::setfill('0') << context.SegEs << "  ";
    std::cout << "FS:  0x" << std::hex << std::setw(4) << std::setfill('0') << context.SegFs << "\n";
    std::cout << "GS:  0x" << std::hex << std::setw(4) << std::setfill('0') << context.SegGs << "  ";
    std::cout << "SS:  0x" << std::hex << std::setw(4) << std::setfill('0') << context.SegSs << "\n\n";

    std::cout << "===== 标志寄存器 =====\n";
    std::cout << "EFlags: 0x" << std::hex << std::setw(8) << std::setfill('0') << context.EFlags << "\n";
}

// 设置硬件断点
bool SetHardwareBreakpoint(HANDLE hThread, uintptr_t address, int type, int size) {
    CONTEXT context = { 0 };
    context.ContextFlags = CONTEXT_DEBUG_REGISTERS | CONTEXT_AMD64; // 显式指定x64

    if (!GetThreadContext(hThread, &context)) {
        std::cerr << "GetThreadContext 失败: " << GetLastError() << "\n";
        return false;
    }

    // 查找空闲调试寄存器 (DR0-DR3)
    int regIndex = -1;
    for (int i = 0; i < 4; ++i) {
        if (!(context.Dr7 & (1ULL << (i * 2)))) {
            regIndex = i;
            break;
        }
    }
    if (regIndex == -1) {
        std::cerr << "无可用调试寄存器（最多4个）\n";
        return false;
    }

    // 设置断点地址
    switch (regIndex) {
    case 0: context.Dr0 = address; break;
    case 1: context.Dr1 = address; break;
    case 2: context.Dr2 = address; break;
    case 3: context.Dr3 = address; break;
    }

    // 启用断点
    context.Dr7 |= (1ULL << (regIndex * 2));

    // 设置触发条件（读/写/读写）
    context.Dr7 &= ~(0x3ULL << (16 + regIndex * 4)); // 清除旧条件
    context.Dr7 |= ((ULONGLONG)type << (16 + regIndex * 4));

    // 设置长度（0=1字节, 1=2字节, 3=4字节, 2=8字节 — 注意 x64 下 8 字节用 2）
    int lenCode = 0;
    if (size == 1) lenCode = 0;
    else if (size == 2) lenCode = 1;
    else if (size == 4) lenCode = 3;
    else if (size == 8) lenCode = 2; // x64: 8字节 = 10b = 2
    else {
        std::cerr << "不支持的断点长度: " << size << "（仅支持1/2/4/8）\n";
        return false;
    }
    context.Dr7 &= ~(0x3ULL << (18 + regIndex * 4)); // 清除旧长度
    context.Dr7 |= ((ULONGLONG)lenCode << (18 + regIndex * 4));

    if (!SetThreadContext(hThread, &context)) {
        std::cerr << "SetThreadContext 失败: " << GetLastError() << "\n";
        return false;
    }
    return true;
}

// 调试事件处理
void HandleDebugEvent(const DEBUG_EVENT& debugEvent) {
    if (debugEvent.dwDebugEventCode == EXCEPTION_DEBUG_EVENT) {
        const EXCEPTION_RECORD& er = debugEvent.u.Exception.ExceptionRecord;
        if (er.ExceptionCode == EXCEPTION_SINGLE_STEP) {
            std::cout << "\n>>> 硬件断点触发 <<<\n";
            std::cout << "触发地址: 0x" << std::hex << (uintptr_t)er.ExceptionAddress << "\n";

            HANDLE hThread = OpenThread(THREAD_GET_CONTEXT, FALSE, debugEvent.dwThreadId);
            if (hThread) {
                CONTEXT ctx = { 0 };
                ctx.ContextFlags = CONTEXT_ALL | CONTEXT_AMD64; // 显式指定x64
                if (GetThreadContext(hThread, &ctx)) {
                    PrintRegisters(ctx);
                }
                CloseHandle(hThread);
            }
        }
    }
}

int main() {
    // 配置控制台编码（支持中文）
    SetConsoleOutputCP(CP_ACP);
    SetConsoleCP(CP_ACP);
    std::ios::sync_with_stdio(false);

    if (!EnableDebugPrivilege()) {
        std::cerr << "警告：提升调试权限失败（请以管理员身份运行）\n";
    }

    std::cout << "=== 64位远程调试器（硬件断点触发） ===\n";
    std::cout << "输入目标进程名（如 notepad.exe）: ";
    std::string procName;
    std::getline(std::cin, procName);

    DWORD pid = GetProcessIdByName(procName);
    if (pid == 0) {
        std::cerr << "错误：未找到进程 \"" << procName << "\"\n";
        return 1;
    }

    std::cout << "找到进程 PID: " << pid << "\n";

    // 附加到进程
    if (!DebugActiveProcess(pid)) {
        std::cerr << "DebugActiveProcess 失败: " << GetLastError() << "\n";
        return 1;
    }

    std::cout << "已附加到进程，请在目标进程中触发内存访问...\n";
    std::cout << "按 Enter 开始监控（或 Ctrl+C 退出）\n";
    std::cin.get();

    // 获取主线程 ID（简化：取第一个线程）
    DWORD mainTid = 0;

    // 先暂停一次以获取初始线程
    DEBUG_EVENT firstEvent;
    if (WaitForDebugEvent(&firstEvent, 1000)) {
        mainTid = firstEvent.dwThreadId;
        ContinueDebugEvent(firstEvent.dwProcessId, firstEvent.dwThreadId, DBG_CONTINUE);
    }
    else {
        // 如果没事件，尝试用 Toolhelp 获取主线程（修复：移除宽字符别名）
        HANDLE hSnap = CreateToolhelp32Snapshot(TH32CS_SNAPTHREAD, 0);
        if (hSnap != INVALID_HANDLE_VALUE) {
            THREADENTRY32 te; // 直接用默认版本（ANSI/Unicode 由项目字符集决定）
            ZeroMemory(&te, sizeof(THREADENTRY32)); // 修复初始化方式
            te.dwSize = sizeof(THREADENTRY32);

            if (Thread32First(hSnap, &te)) { // 移除 W 后缀
                do {
                    if (te.th32OwnerProcessID == pid) {
                        mainTid = te.th32ThreadID;
                        break;
                    }
                } while (Thread32Next(hSnap, &te)); // 移除 W 后缀
            }
            CloseHandle(hSnap);
        }
    }

    if (mainTid == 0) {
        std::cerr << "无法获取主线程 ID\n";
        DebugActiveProcessStop(pid);
        return 1;
    }

    // 增加线程权限，避免操作失败
    HANDLE hMainThread = OpenThread(
        THREAD_SET_CONTEXT | THREAD_GET_CONTEXT | THREAD_SUSPEND_RESUME,
        FALSE,
        mainTid
    );
    if (!hMainThread) {
        std::cerr << "OpenThread 失败: " << GetLastError() << "\n";
        DebugActiveProcessStop(pid);
        return 1;
    }

    // 设置断点
    std::cout << "\n选择触发类型:\n";
    std::cout << "1. 读取触发\n2. 写入触发\n3. 读写触发\n> ";
    int trigType;
    std::cin >> trigType;

    if (trigType < 1 || trigType > 3) {
        std::cerr << "无效选择\n";
        CloseHandle(hMainThread);
        DebugActiveProcessStop(pid);
        return 1;
    }

    std::cout << "输入触发地址（十六进制，如 7ff6a1b2c3d4）: 0x";
    uintptr_t addr;
    std::cin >> std::hex >> addr;

    std::cout << "输入长度（1/2/4/8 字节）: ";
    int len;
    std::cin >> len;

    // 映射到 DR7 条件码：0=执行, 1=写入, 3=读写
    // 注意：x86/x64 硬件断点不支持纯"读取"，只有写入(1)和读写(3)
    int dr7Type = (trigType == 2) ? 1 : 3; // 2=写入, 其他=读写

    if (!SetHardwareBreakpoint(hMainThread, addr, dr7Type, len)) {
        std::cerr << "设置硬件断点失败\n";
        CloseHandle(hMainThread);
        DebugActiveProcessStop(pid);
        return 1;
    }

    std::cout << "[OK] 断点设置成功！等待触发...\n";

    // 调试循环
    DEBUG_EVENT de;
    while (WaitForDebugEvent(&de, INFINITE)) {
        HandleDebugEvent(de);
        ContinueDebugEvent(de.dwProcessId, de.dwThreadId, DBG_CONTINUE);
    }

    // 清理资源
    CloseHandle(hMainThread);
    DebugActiveProcessStop(pid);
    std::cout << "\n调试结束。\n";
    return 0;
}
